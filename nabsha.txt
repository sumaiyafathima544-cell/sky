<script>
function showWeather(){
  const city = document.getElementById("city").value;
  const date = document.getElementById("date").value;
  const time = document.getElementById("time").value;

  if(city==="" || date==="" || time===""){
    alert("City, Date, Time ellam enter pannunga");
    return;
  }

  // Selected time
  const t1 = new Date(date + "T" + time);
  const t2 = new Date(t1.getTime() + 30 * 60000); // Next half-hour

  // Temperature logic: simple simulation
  function getTemp(hour, seed){
    // seed randomization for city + hour for consistent temp
    const hash = city.charCodeAt(0) + seed;
    return Math.round(26 + Math.sin(hour/24 * Math.PI*2 + hash) * 5);
  }

  // Weather condition logic: randomize next half-hour
  function getCondition(hour, seed){
    const options = ["Sunny ‚òÄÔ∏è","Rainy üåßÔ∏è","Cloudy ‚õÖ","Windy üå¨Ô∏è"];
    const index = (hour + seed) % options.length;
    return options[index];
  }

  const seed = t1.getDate(); // unique per day
  // Selected time weather
  document.getElementById("time1").innerText = 
      city + " | " + t1.toLocaleTimeString([], {hour:'2-digit', minute:'2-digit'});
  document.getElementById("temp1").innerText = getTemp(t1.getHours(), seed) + "¬∞C";
  document.getElementById("cond1").innerText = getCondition(t1.getHours(), seed);

  // Next half-hour weather
  document.getElementById("time2").innerText = 
      city + " | " + t2.toLocaleTimeString([], {hour:'2-digit', minute:'2-digit'});
  document.getElementById("temp2").innerText = getTemp(t2.getHours(), seed+1) + "¬∞C";
  document.getElementById("cond2").innerText = getCondition(t2.getHours(), seed+1);

  document.getElementById("result").style.display = "block";
 
  }
</script>
