/*

let lastWeather = null;

// Map Open-Meteo weathercode to a friendly condition
// Reference: https://open-meteo.com/en/docs
const weatherCodeMap = [
  {codes:[0], name:'Clear', emoji:'‚òÄÔ∏è'},
  {codes:[1,2,3], name:'Mainly clear / partly cloudy', emoji:'üå§Ô∏è'},
  {codes:[45,48], name:'Fog', emoji:'üå´Ô∏è'},
  {codes:[51,53,55,56,57], name:'Drizzle / Freezing Drizzle', emoji:'üå¶Ô∏è'},
  {codes:[61,63,65,66,67], name:'Rain', emoji:'üåßÔ∏è'},
  {codes:[71,73,75,77], name:'Snow', emoji:'‚ùÑÔ∏è'},
  {codes:[80,81,82], name:'Rain showers', emoji:'üåßÔ∏è'},
  {codes:[95,96,99], name:'Thunderstorm', emoji:'‚õàÔ∏è'},
  // fallback
];

function mapWeatherCode(code){
  for(const entry of weatherCodeMap){
    if(entry.codes.includes(code)) return entry;
  }
  return {name:'Unknown', emoji:'üåà'};
}

// Challenge rules: simple, unique, actionable
function pickChallenge(conditionName, temperature){
  // default challenges
  const lowTemp = temperature !== null && temperature <= 15;
  const highTemp = temperature !== null && temperature >= 30;

  const pool = [];

  // Clear / sunny
  if(/clear|sun/i.test(conditionName)){
    pool.push(
      {title:'Sunshine Walk', text:'Take a 15-minute walk outside and notice 3 things you like about the sky.'},
      {title:'Sun Gratitude', text:'Step outside for 5 minutes and take 3 deep breaths ‚Äî enjoy the sunlight.'}
    );
  }

  // Rainy / drizzle
  if(/rain|drizzle|showers/i.test(conditionName)){
    pool.push(
      {title:'Rain Reading', text:'Spend 20 minutes reading or listening to music while sipping a warm drink.'},
      {title:'Cozy Care', text:'Do one cozy task: warm socks, a hot drink, or a short stretch.'}
    );
  }

  // Cloudy / overcast
  if(/cloud|fog|mist|mainly/i.test(conditionName)){
    pool.push(
      {title:'Photo Challenge', text:'Take a moody photo of the sky and save it to your gallery.'},
      {title:'Mindful Pause', text:'Close your eyes for 2 minutes and do a quick body-scan.'}
    );
  }

  // Thunderstorm
  if(/thunder|storm/i.test(conditionName)){
    pool.push(
      {title:'Safety Check', text:'Quickly check that windows are closed and devices are safe from power surges.'},
      {title:'Indoor Stretch', text:'Do a 7-minute indoor stretch routine to stay active.'}
    );
  }

  // Snow/cold
  if(/snow|freez/i.test(conditionName) || lowTemp){
    pool.push(
      {title:'Warm Note', text:'Make a hot beverage and write a 1-line note of appreciation to yourself.'},
      {title:'Warm-Up Move', text:'Do 10 simple bodywarmers: jumping jacks or marching on the spot.'}
    );
  }

  // Hot day
  if(highTemp){
    pool.push(
      {title:'Hydration Break', text:'Drink a full glass of water and set a 1-hour reminder to hydrate again.'},
      {title:'Shade Walk', text:'Take a 10-minute walk in the shade or indoors to stay cool.'}
    );
  }

  // Default gentle ideas
  pool.push(
    {title:'Tiny Task', text:'Complete one small task you‚Äôve been postponing (5‚Äì15 minutes).'},
    {title:'Smile Challenge', text:'Send a short message to someone and make their day.'}
  );

  // pick one randomly
  const choice = pool[Math.floor(Math.random()*pool.length)];
  return choice;
}

async function fetchWeatherByCoords(lat, lon){
  try{
    miniLoader.style.display = 'inline-block';
    statusEl.textContent = 'Fetching current weather‚Ä¶';
    const url = `https://api.open-meteo.com/v1/forecast?latitude=${encodeURIComponent(lat)}&longitude=${encodeURIComponent(lon)}&current_weather=true&timezone=auto`;
    const resp = await fetch(url);
    if(!resp.ok) throw new Error('Weather fetch failed');
    const data = await resp.json();
    miniLoader.style.display = 'none';
    if(!data || !data.current_weather) throw new Error('No weather data');

    const cw = data.current_weather;
    const temperature = Math.round(cw.temperature); // in ¬∞C
    const code = cw.weathercode;
    const condition = mapWeatherCode(code);

    // display
    tempEl.textContent = `${temperature}¬∞C`;
    condEl.textContent = `${condition.name} ‚Ä¢ wind ${Math.round(cw.windspeed)} km/h`;
    emojiEl.textContent = condition.emoji;
    placeEl.textContent = `${data.timezone || ''}`;
    const now = new Date();
    timeLocalEl.textContent = now.toLocaleString();

    // pick challenge
    const challenge = pickChallenge(condition.name, temperature);
    challengeTitleEl.textContent = challenge.title;
    challengeTextEl.textContent = challenge.text;
    challengeBox.style.display = 'flex';
    statusEl.textContent = 'Today‚Äôs challenge ready ‚Äî go for it!';
    lastWeather = {temperature, code, condition};
  }catch(err){
    miniLoader.style.display = 'none';
    statusEl.textContent = 'Unable to fetch weather. Try search or refresh.';
    condEl.textContent = '‚Äî';
    tempEl.textContent = '--¬∞C';
    console.error(err);
  }
}

async function geocodeAndFetch(city){
  if(!city || !city.trim()) return;
  statusEl.textContent = `Searching for "${city}"‚Ä¶`;
  try{
    const geoUrl = `https://geocoding-api.open-meteo.com/v1/search?name=${encodeURIComponent(city)}&count=5&language=en&format=json`;
    const r = await fetch(geoUrl);
    if(!r.ok) throw new Error('Geocoding failed');
    const j = await r.json();
    if(!j.results || j.results.length === 0){
      statusEl.textContent = 'City not found. Try another name.';
      return;
    }
    const top = j.results[0];
    placeEl.textContent = `${top.name}${top.admin1 ? ', ' + top.admin1 : ''}${top.country ? ', ' + top.country : ''}`;
    await fetchWeatherByCoords(top.latitude, top.longitude);
  }catch(e){
    console.error(e);
    statusEl.textContent = 'Geocoding failed.';
  }
}

function tryGeolocation(){
  if(!navigator.geolocation) {
    statusEl.textContent = 'Geolocation not supported ‚Äî search a city.';
    return;
  }
  statusEl.textContent = 'Requesting location‚Ä¶';
  navigator.geolocation.getCurrentPosition(async (pos)=>{
    const lat = pos.coords.latitude, lon = pos.coords.longitude;
    statusEl.textContent = 'Location found ‚Äî fetching weather‚Ä¶';
    // Optionally reverse-geocode the display name via Open-Meteo geocoding reverse endpoint
    try{
      const rev = await fetch(`https://geocoding-api.open-meteo.com/v1/reverse?latitude=${lat}&longitude=${lon}&count=1`);
      if(rev.ok){
        const j = await rev.json();
        if(j && j.results && j.results.length) {
          const r = j.results[0];
          placeEl.textContent = `${r.name}${r.admin1 ? ', ' + r.admin1 : ''}${r.country ? ', ' + r.country : ''}`;
        } else {
          placeEl.textContent = `${lat.toFixed(2)}, ${lon.toFixed(2)}`;
        }
      } else {
        placeEl.textContent = `${lat.toFixed(2)}, ${lon.toFixed(2)}`;
      }
    }catch(e){
      placeEl.textContent = `${lat.toFixed(2)}, ${lon.toFixed(2)}`;
    }
    await fetchWeatherByCoords(lat, lon);
  }, (err)=>{
    console.warn('Geolocation error', err);
    statusEl.textContent = 'Location blocked or unavailable ‚Äî search a city.';
  }, {timeout:8000, maximumAge:300000});
}

// UI events
searchBtn.addEventListener('click', ()=> geocodeAndFetch(cityInput.value));
cityInput.addEventListener('keyup', (e)=> { if(e.key === 'Enter') geocodeAndFetch(cityInput.value); });
refreshBtn.addEventListener('click', ()=> {
  statusEl.textContent = 'Refreshing‚Ä¶';
  if(lastWeather) {
    // try to refresh using the last known timezone/place by attempt geolocation first
    tryGeolocation();
  } else {
    tryGeolocation();
  }
});
doneBtn.addEventListener('click', ()=> {
  challengeTipEl.textContent = 'Great! Marked done ‚úÖ ‚Äî come back tomorrow for a new challenge.';
  doneBtn.disabled = true;
  doneBtn.style.opacity = 0.6;
  statusEl.textContent = 'Challenge completed ‚Äî nice one!';
});

// start
tryGeolocation();
